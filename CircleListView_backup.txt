// MARK: - Dominant Color Extraction

extension UIImage {
    /// Returns the most frequent color (ignoring near-black/white pixels).
    var dominantColor: UIColor? {
        let sampleSize = CGSize(width: 50, height: 50)
        UIGraphicsBeginImageContextWithOptions(sampleSize, true, 1)
        defer { UIGraphicsEndImageContext() }
        draw(in: CGRect(origin: .zero, size: sampleSize))
        guard let small = UIGraphicsGetImageFromCurrentImageContext(),
              let cg = small.cgImage,
              let data = cg.dataProvider?.data,
              let ptr = CFDataGetBytePtr(data) else { return nil }

        let bpp = cg.bitsPerPixel / 8
        let bpr = cg.bytesPerRow
        let w = cg.width, h = cg.height

        let passes: [(lo: Int, hi: Int)] = [(25, 230), (0, 255)]
        var buckets: [UInt32: (n: Int, r: Int, g: Int, b: Int)] = [:]

        for pass in passes {
            if !buckets.isEmpty { break }
            for y in 0..<h {
                for x in 0..<w {
                    let off = y * bpr + x * bpp
                    let r = Int(ptr[off]), g = Int(ptr[off + 1]), b = Int(ptr[off + 2])
                    let lum = (r + g + b) / 3
                    if lum < pass.lo || lum > pass.hi { continue }
                    let key = (UInt32(r / 16) << 16) | (UInt32(g / 16) << 8) | UInt32(b / 16)
                    if var bk = buckets[key] {
                        bk.n += 1; bk.r += r; bk.g += g; bk.b += b
                        buckets[key] = bk
                    } else {
                        buckets[key] = (n: 1, r: r, g: g, b: b)
                    }
                }
            }
        }

        guard let best = buckets.values.max(by: { $0.n < $1.n }) else { return nil }
        return UIColor(red: CGFloat(best.r) / CGFloat(best.n) / 255,
                       green: CGFloat(best.g) / CGFloat(best.n) / 255,
                       blue: CGFloat(best.b) / CGFloat(best.n) / 255,
                       alpha: 1)
    }
}

// MARK: - Carousel Metrics

private enum CarouselMetrics {
    static let cardSpacing: CGFloat = 12
    /// Card width: ~2/3 screen so 1/3 of the next card peeks
    static var cardWidth: CGFloat {
        (UIScreen.main.bounds.width - 40) * 0.72
    }
    static let cardCornerRadius: CGFloat = 16
    static let accentAvatarSize: CGFloat = 24
    static let cardBodyPadding: CGFloat = 14
    /// Image is square, ~75% of card width
    static var imageSize: CGFloat {
        cardWidth * 0.75
    }
}

// MARK: - Circle List View

struct CircleListView: View {
    let circles: [TallyCircle]
    @Binding var searchText: String
    var onAdd: () -> Void
    var recentActivityText: String?

    @State private var scrollOffset: CGFloat = 0
    @State private var showSearch = false
    @State private var favoriteIDs: Set<String> = FavoriteCircleStore.loadAll()
    @State private var currentCardIndex: Int = 0

    private var headerOpacity: Double {
        let fadeStart: CGFloat = 40
        let fadeEnd: CGFloat = 120
        if scrollOffset <= fadeStart { return 1 }
        if scrollOffset >= fadeEnd { return 0 }
        return Double(1 - (scrollOffset - fadeStart) / (fadeEnd - fadeStart))
    }

    private var filteredCircles: [TallyCircle] {
        let filtered: [TallyCircle]
        if searchText.isEmpty {
            filtered = circles
        } else {
            filtered = circles.filter {
                $0.name.localizedCaseInsensitiveContains(searchText) ||
                $0.members.contains { $0.name.localizedCaseInsensitiveContains(searchText) }
            }
        }
        return filtered.sorted { a, b in
            let aFav = a.serverId.map { favoriteIDs.contains($0) } ?? false
            let bFav = b.serverId.map { favoriteIDs.contains($0) } ?? false
            if aFav != bFav { return aFav }
            return false
        }
    }

    private var topPadding: CGFloat {
        showSearch ? 120 : 70
    }

    var body: some View {
        ZStack(alignment: .top) {
            ScrollView(.vertical, showsIndicators: false) {
                VStack(alignment: .leading, spacing: 0) {
                    Color.clear.frame(height: topPadding)

                    // Notification banner
                    notificationBanner
                        .padding(.horizontal, TallySpacing.screenPadding)
                        .padding(.bottom, TallySpacing.xxl)

                    // Section header
                    sectionHeader
                        .padding(.horizontal, TallySpacing.screenPadding)
                        .padding(.bottom, TallySpacing.lg)

                    // Card carousel
                    cardCarousel
                }
            }
            .contentMargins(.bottom, 100, for: .scrollContent)
            .onScrollGeometryChange(for: CGFloat.self) { geo in
                geo.contentOffset.y
            } action: { _, newValue in
                scrollOffset = newValue
            }

            // Floating header
            floatingHeader
                .opacity(headerOpacity)
                .allowsHitTesting(headerOpacity > 0.1)
        }
        .background(Color.white.ignoresSafeArea())
    }

    // MARK: - Floating Header

    private var floatingHeader: some View {
        VStack(alignment: .leading, spacing: TallySpacing.sm) {
            HStack(alignment: .center) {
                Text("Circles")
                    .font(TallyFont.largeTitle)
                    .foregroundStyle(TallyColors.textPrimary)

                Spacer()

                HStack(spacing: TallySpacing.sm) {
                    Button {
                        withAnimation(.snappy(duration: 0.25)) {
                            showSearch.toggle()
                            if !showSearch { searchText = "" }
                        }
                    } label: {
                        Image(systemName: "magnifyingglass")
                            .font(TallyIcon.lg)
                            .foregroundStyle(TallyColors.ink)
                            .frame(width: 44, height: 44)
                            .liquidGlass(in: Circle())
                    }

                    Button(action: onAdd) {
                        Image(systemName: "plus")
                            .font(TallyIcon.lg)
                            .foregroundStyle(TallyColors.ink)
                            .frame(width: 44, height: 44)
                            .liquidGlass(in: Circle())
                    }
                }
            }

            if showSearch {
                searchBar
                    .transition(.move(edge: .top).combined(with: .opacity))
            }
        }
        .padding(.horizontal, TallySpacing.screenPadding)
        .padding(.top, TallySpacing.sm)
        .padding(.bottom, TallySpacing.sm)
        .background(
            LinearGradient(
                stops: [
                    .init(color: Color.white.opacity(0.9), location: 0),
                    .init(color: Color.white.opacity(0.6), location: 0.35),
                    .init(color: Color.white.opacity(0.2), location: 0.7),
                    .init(color: Color.white.opacity(0), location: 1.0),
                ],
                startPoint: .top,
                endPoint: .bottom
            )
        )
    }

    // MARK: - Search Bar

    private var searchBar: some View {
        HStack(spacing: TallySpacing.sm) {
            Image(systemName: "magnifyingglass")
                .font(TallyIcon.md)
                .foregroundStyle(TallyColors.textSecondary)

            TextField("Search circles", text: $searchText)
                .font(TallyFont.body)
                .foregroundStyle(TallyColors.textPrimary)

            if !searchText.isEmpty {
                Button {
                    searchText = ""
                } label: {
                    Image(systemName: "xmark.circle.fill")
                        .font(TallyIcon.md)
                        .foregroundStyle(TallyColors.textSecondary)
                }
            }
        }
        .padding(.horizontal, TallySpacing.lg)
        .frame(height: 44)
        .liquidGlass(in: Capsule())
    }

    // MARK: - Notification Banner

    private var notificationBanner: some View {
        HStack(spacing: TallySpacing.lg) {
            if recentActivityText != nil {
                Image(systemName: "bell.badge.fill")
                    .font(TallyIcon.xxl)
                    .foregroundStyle(TallyColors.textPrimary)
                    .frame(width: 50, height: 50)
            } else {
                HStack(spacing: 1) {
                    Image(systemName: "apple.logo")
                        .font(.system(size: 18, weight: .medium))
                        .offset(y: -1)
                    Text("Pay")
                        .font(.system(size: 20, weight: .medium))
                }
                .foregroundStyle(TallyColors.textPrimary)
                .padding(.horizontal, 10)
                .padding(.vertical, 8)
                .overlay(
                    RoundedRectangle(cornerRadius: 6)
                        .stroke(TallyColors.textPrimary, lineWidth: 1.5)
                )
            }

            VStack(alignment: .leading, spacing: 5) {
                Text(recentActivityText ?? "Add Cards To Your Apple Wallet")
                    .font(TallyFont.bodyInput)
                    .foregroundStyle(TallyColors.textPrimary)
                    .lineLimit(2)
                Text(recentActivityText != nil ? "Just now" : "Tap to set up contactless pay")
                    .font(TallyFont.bodySmall)
                    .foregroundStyle(TallyColors.textSecondary)
                    .lineLimit(1)
            }

            Spacer()

            Image(systemName: "chevron.right")
                .font(TallyIcon.sm)
                .fontWeight(.semibold)
                .foregroundStyle(TallyColors.textTertiary)
        }
        .padding(.horizontal, TallySpacing.xl)
        .padding(.vertical, 24)
        .background(TallyColors.bgPrimary)
        .clipShape(RoundedRectangle(cornerRadius: 16))
        .shadow(color: .black.opacity(0.08), radius: 12, y: 4)
        .shadow(color: .black.opacity(0.04), radius: 2, y: 1)
    }

    // MARK: - Section Header

    private var sectionHeader: some View {
        HStack {
            HStack(spacing: 8) {
                Text("Your Circles")
                    .font(TallyFont.brandCard)
                    .foregroundStyle(TallyColors.textPrimary)

                Text("\(filteredCircles.count)")
                    .font(TallyFont.overline)
                    .foregroundStyle(TallyColors.textSecondary)
            }

            Spacer()

            // Dot indicators (max 6)
            HStack(spacing: 5) {
                ForEach(0..<min(filteredCircles.count, 6), id: \.self) { i in
                    Capsule()
                        .fill(i == currentCardIndex ? TallyColors.textPrimary : TallyColors.textTertiary.opacity(0.35))
                        .frame(width: i == currentCardIndex ? 18 : 8, height: 8)
                        .animation(.snappy(duration: 0.25), value: currentCardIndex)
                }
            }
        }
    }

    // MARK: - Card Carousel

    private var cardCarousel: some View {
        ScrollView(.horizontal, showsIndicators: false) {
            LazyHStack(spacing: CarouselMetrics.cardSpacing) {
                ForEach(Array(filteredCircles.enumerated()), id: \.element.id) { index, circle in
                    NavigationLink(value: circle) {
                        CarouselCircleCard(
                            circle: circle,
                            index: index,
                            favoriteIDs: $favoriteIDs
                        )
                    }
                    .buttonStyle(.plain)
                }
            }
            .padding(.horizontal, TallySpacing.screenPadding)
        }
        .scrollClipDisabled()
        .onScrollGeometryChange(for: CGFloat.self) { geo in
            geo.contentOffset.x
        } action: { _, newX in
            let cardPlusSpacing = CarouselMetrics.cardWidth + CarouselMetrics.cardSpacing
            let idx = Int(round(newX / cardPlusSpacing))
            currentCardIndex = max(0, min(idx, filteredCircles.count - 1))
        }
    }
}

// MARK: - Carousel Circle Card

private struct CarouselCircleCard: View {
    let circle: TallyCircle
    let index: Int
    @Binding var favoriteIDs: Set<String>

    private var themeUIColor: UIColor {
        if let photo = circle.photo, let dom = photo.dominantColor { return dom }
        return UIColor(TallyColors.cardColor(for: index))
    }

    private var themeColor: Color { Color(themeUIColor) }

    /// Blends themeColor with white at the given strength (opaque).
    private func tint(_ strength: CGFloat) -> Color {
        var r: CGFloat = 0, g: CGFloat = 0, b: CGFloat = 0
        themeUIColor.getRed(&r, green: &g, blue: &b, alpha: nil)
        return Color(red: Double(r * strength + (1 - strength)),
                     green: Double(g * strength + (1 - strength)),
                     blue: Double(b * strength + (1 - strength)))
    }

    /// Darkens the theme color (blends toward black).
    private func shade(_ strength: CGFloat) -> Color {
        var r: CGFloat = 0, g: CGFloat = 0, b: CGFloat = 0
        themeUIColor.getRed(&r, green: &g, blue: &b, alpha: nil)
        return Color(red: Double(r * strength),
                     green: Double(g * strength),
                     blue: Double(b * strength))
    }

    private var isFavorited: Bool {
        guard let sid = circle.serverId else { return false }
        return favoriteIDs.contains(sid)
    }

    private var latestActivityText: String {
        if let latest = circle.transactions
            .sorted(by: { $0.date > $1.date })
            .first {
            return "\(latest.emoji) \(latest.paidBy) spent $\(String(format: "%.0f", latest.amount)) on \(latest.title)"
        }
        return "No activity yet"
    }

    var body: some View {
        VStack(spacing: 0) {
            // ── Accent strip ──
            accentStrip

            // ── Card body ──
            VStack(alignment: .leading, spacing: 0) {
                // Centered image
                HStack {
                    Spacer()
                    imageArea
                    Spacer()
                }
                .padding(.top, 16)
                .padding(.bottom, 20)

                // Title
                Text(circle.name)
                    .font(TallyFont.brandCard)
                    .foregroundStyle(TallyColors.textPrimary)
                    .lineLimit(2)
                    .padding(.horizontal, CarouselMetrics.cardBodyPadding)

                // Balance + member count
                HStack(spacing: 8) {
                    Text(formattedBalance)
                        .font(TallyFont.brandCard)
                        .foregroundStyle(TallyColors.textPrimary)

                    HStack(spacing: 3) {
                        Image(systemName: "person.fill")
                            .font(TallyIcon.xxs)
                        Text("\(circle.memberCount)")
                    }
                    .font(TallyFont.smallSemibold)
                    .foregroundStyle(TallyColors.textSecondary)
                }
                .padding(.horizontal, CarouselMetrics.cardBodyPadding)
                .padding(.top, 4)

                // Latest activity
                Text(latestActivityText)
                    .font(TallyFont.caption)
                    .foregroundStyle(TallyColors.textSecondary)
                    .lineLimit(1)
                    .padding(.horizontal, CarouselMetrics.cardBodyPadding)
                    .padding(.top, 6)
                    .padding(.bottom, 14)

                // Actions row
                actionsRow
                    .padding(.horizontal, CarouselMetrics.cardBodyPadding)
                    .padding(.bottom, CarouselMetrics.cardBodyPadding)
            }
            .background(
                ZStack {
                    // Base gradient: themed dark → near white
                    LinearGradient(
                        stops: [
                            .init(color: shade(0.4), location: 0),
                            .init(color: tint(0.25), location: 0.3),
                            .init(color: tint(0.08), location: 0.6),
                            .init(color: Color(white: 0.97), location: 1.0),
                        ],
                        startPoint: .top,
                        endPoint: .bottom
                    )
                    // Nebula glow: radial highlight for depth
                    RadialGradient(
                        colors: [themeColor.opacity(0.2), .clear],
                        center: .topLeading,
                        startRadius: 10,
                        endRadius: CarouselMetrics.cardWidth * 0.7
                    )
                }
            )
        }
        .frame(width: CarouselMetrics.cardWidth)
        .clipShape(RoundedRectangle(cornerRadius: CarouselMetrics.cardCornerRadius))
        .overlay(
            RoundedRectangle(cornerRadius: CarouselMetrics.cardCornerRadius)
                .strokeBorder(Color.black.opacity(0.08), lineWidth: 0.5)
        )
        .shadow(color: .black.opacity(0.15), radius: 20, y: 8)
        .shadow(color: .black.opacity(0.06), radius: 4, y: 2)
    }

    // MARK: - Accent Strip

    private var accentStrip: some View {
        HStack(spacing: 8) {
            if let photo = circle.photo {
                Image(uiImage: photo)
                    .resizable()
                    .scaledToFill()
                    .frame(width: CarouselMetrics.accentAvatarSize,
                           height: CarouselMetrics.accentAvatarSize)
                    .clipShape(RoundedRectangle(cornerRadius: 6))
            } else {
                Text(String(circle.name.prefix(1)).uppercased())
                    .font(TallyFont.microBold)
                    .foregroundStyle(.white)
                    .frame(width: CarouselMetrics.accentAvatarSize,
                           height: CarouselMetrics.accentAvatarSize)
                    .background(.white.opacity(0.25))
                    .clipShape(RoundedRectangle(cornerRadius: 6))
            }

            Text(circle.name)
                .font(TallyFont.buttonBold)
                .foregroundStyle(.white)
                .lineLimit(1)

            Spacer()

            Image(systemName: "chevron.right")
                .font(TallyIcon.xs)
                .foregroundStyle(.white.opacity(0.5))
        }
        .padding(.horizontal, CarouselMetrics.cardBodyPadding)
        .padding(.vertical, 6)
        .background(shade(0.5))
    }

    // MARK: - Image Area

    private var imageArea: some View {
        Group {
            if let photo = circle.photo {
                Image(uiImage: photo)
                    .resizable()
                    .scaledToFill()
                    .frame(width: CarouselMetrics.imageSize, height: CarouselMetrics.imageSize)
                    .clipped()
                    .shadow(color: .black.opacity(0.12), radius: 4, y: 2)
            } else {
                photoFallback
            }
        }
    }

    // MARK: - Photo Fallback

    private var photoFallback: some View {
        VStack(spacing: 12) {
            Text(String(circle.name.prefix(1)).uppercased())
                .font(TallyFont.brandHero)
                .foregroundStyle(themeColor.opacity(0.25))

            HStack(spacing: -8) {
                ForEach(Array(circle.members.prefix(4))) { member in
                    Text(member.initial)
                        .font(TallyFont.buttonBold)
                        .foregroundStyle(.white)
                        .frame(width: 36, height: 36)
                        .background(member.color.opacity(0.6))
                        .clipShape(Circle())
                }
            }
        }
        .frame(width: CarouselMetrics.imageSize, height: CarouselMetrics.imageSize)
        .background(shade(0.25))
    }

    // MARK: - Actions Row

    private var actionsRow: some View {
        HStack(spacing: 0) {
            memberAvatarRow

            Spacer()

            Button { toggleFavorite() } label: {
                Image(systemName: isFavorited ? "star.fill" : "star")
                    .font(TallyIcon.xl)
                    .foregroundStyle(isFavorited ? .yellow : TallyColors.textPrimary)
            }
            .buttonStyle(.plain)

            ShareLink(
                item: "Join my \"\(circle.name)\" circle on Mntly!",
                subject: Text("Join \(circle.name)"),
                message: Text("Split expenses with friends")
            ) {
                Image(systemName: "arrowshape.turn.up.right.fill")
                    .font(TallyIcon.xl)
                    .foregroundStyle(TallyColors.textPrimary)
            }
            .padding(.leading, TallySpacing.md)
        }
    }

    // MARK: - Helpers

    private var formattedBalance: String {
        String(format: "$%.2f", circle.walletBalance)
    }

    private func toggleFavorite() {
        guard let sid = circle.serverId else { return }
        if favoriteIDs.contains(sid) {
            favoriteIDs.remove(sid)
        } else {
            favoriteIDs.insert(sid)
        }
        FavoriteCircleStore.save(favoriteIDs)
    }

    @ViewBuilder
    private var memberAvatarRow: some View {
        HStack(spacing: -8) {
            avatarBubble(initial: "Y", color: TallyColors.accent)
            ForEach(Array(circle.members.prefix(3))) { member in
                avatarBubble(initial: member.initial, color: member.color)
            }
            if circle.members.count > 3 {
                Text("+\(circle.members.count - 3)")
                    .font(TallyFont.smallLabel)
                    .foregroundStyle(TallyColors.textSecondary)
                    .frame(width: 28, height: 28)
                    .background(TallyColors.bgSecondary, in: Circle())
            }
        }
    }

    private func avatarBubble(initial: String, color: Color) -> some View {
        Text(initial)
            .font(TallyFont.captionBold)
            .foregroundStyle(.white)
            .frame(width: 28, height: 28)
            .background(color)
            .clipShape(Circle())
    }
}
